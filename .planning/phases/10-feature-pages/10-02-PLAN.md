```typescript
function taskStatusBadgeClass(status: string): string {
  switch (status) {
    case "in_progress": return "bg-yellow-900/40 text-yellow-400";
    case "complete":    return "bg-green-900/40 text-[#4ec994]";
    default:            return "bg-[#2a2d2e] text-[#858585]";
  }
}

function taskStatusLabel(status: string): string {
  switch (status) {
    case "in_progress": return "In progress";
    case "complete":    return "Complete";
    default:            return "Open";
  }
}
```

Component structure for `export function QuickTasksPage()`:

1. Context: `const { activeSegment, loading } = useApp();`
2. State:
   - `const [tasks, setTasks] = useState<QuickTaskPayload[]>([]);`
   - `const [fetchLoading, setFetchLoading] = useState(false);`
   - `const [error, setError] = useState<string | null>(null);`

3. Fetch effect (QTSK-01):
```typescript
useEffect(() => {
  if (!activeSegment?.planningPath) return;
  setFetchLoading(true);
  void fetchQuickTasks(activeSegment.planningPath)
    .then((t) => { setTasks(t); setError(null); })
    .catch(() => setError("Could not load quick tasks. Check your project path and try refreshing."))
    .finally(() => setFetchLoading(false));
}, [activeSegment?.planningPath]);
```

4. Sort by last_updated descending (QTSK-05):
```typescript
const sorted = useMemo(
  () =>
    [...tasks].sort((a, b) => {
      if (a.last_updated && b.last_updated)
        return b.last_updated.localeCompare(a.last_updated);
      if (a.last_updated) return -1;
      if (b.last_updated) return 1;
      return 0;
    }),
  [tasks],
);
```

5. Guards (in this exact order):
   - `if (loading || fetchLoading) return <div className="p-6 text-[#858585] text-sm">Loading\u2026</div>;`
   - `if (!activeSegment) return <div className="p-6 text-[#858585] text-sm">Add scan roots in Settings and select a project.</div>;`
   - `if (error) return <div className="p-6"><p className="text-red-400 text-sm">{error}</p></div>;`
   - Empty state (QTSK-04): `if (sorted.length === 0) return (<div className="p-6 text-center text-[#858585] text-sm"><p className="font-medium">No quick tasks yet</p><p>Quick tasks appear here when you run /gsd:quick in this project.</p></div>);`

6. Main return — task list (QTSK-02, QTSK-03):

```
<div className="p-6 space-y-2">
```
